· 2023-12-10 ·

类似 dp，预处理的时候定义 dp[i][j] 为从 iii 开始的 2j2^j2j 个元素中的最大的。

得出 dpi,j=max⁡(dpi,j−1,dpi+2j−1,j−1)dp_{i,j} = \max(dp_{i, j - 1}, dp_{i + 2 ^{j - 1}, j- 1})dpi,j​=max(dpi,j−1​,dpi+2j−1,j−1​)。

##代码

#include <bits/stdc++.h>

using namespace std;
/*
*/
int n, m, a[1000005], dp[100005][25];
int main() {
    ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
    cin >> n >> m;
    for (int i = 1; i <= n; i ++) {
        cin >> a[i];
        dp[i][0] = a[i];
    } 
    for (int i = 1; i <= 21; i ++) {
        for (int j = 1; j + (1 << i) - 1 <= n; j ++) {
            dp[j][i] = max(dp[j][i - 1], dp[j + (1 << (i - 1))][i - 1]);
        }
    }
    while (m --) {
        int l, r; cin >> l >> r;
        int c = __lg(r - l + 1);
        cout << max(dp[l][c], dp[r - (1 << c) + 1][c]) << "\n";
    }
    return 0;
}
